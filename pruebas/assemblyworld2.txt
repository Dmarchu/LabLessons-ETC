        .globl __start
	.data 0x10000000
hello:  .asciiz "Payica infiel"
req:    .asciiz "Iteraciones: "

	.text 0x00400000
__start:
	la $a0, req
	jal print
	li $v0, 5
	syscall
	move $a1, $v0
	jal line
	li $a2, 0
	j loop
line:
	li $a0, 10
	li $v0, 11
	syscall
	jr $ra
loop:
	beq $a1, $a2, end
	la $a0, hello
	jal print
	jal line
	addi $a2, $a2, 1
	j loop
print:
	li $v0, 4
	syscall
	jr $ra
end:    
	li $v0, 10
	syscall
